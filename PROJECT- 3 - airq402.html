# Method to identify and replace outliers with mean
def removeOutliers(x, column_name):
    a = np.array(x)
    q3 = np.percentile(a, 75)
    q1 = np.percentile(a, 25)
    iqr = q3 - q1
    quartileSet = [q1 - 1.5*iqr, q3 + 1.5*iqr]
    mean = x.mean()
    resultList = []
    for i in range(len(a)):
        if a[i] <= quartileSet[0] or a[i] >= quartileSet[1]:
            resultList.append(a[i])
            a[i] = mean
    print("---***---")
    print("COLUMN NAME: ", column_name)
    print("Quartile Set: ", quartileSet)
    print("Mean: ", mean)
    print("Number of outliers: ", len(resultList))
    print("Outliers: ", resultList)
    # x = pd.Series(a)
    return pd.Series(a)

df_airq['Distance'] = removeOutliers(df_airq['Distance'], 'Distance')
