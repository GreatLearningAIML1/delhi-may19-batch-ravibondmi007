TRAIN_CSV = 'train_labels.csv'

IMAGE_SIZE = 128 # MobileNet takes images of size 128*128*3 

with open(TRAIN_CSV, 'r') as csvfile:
    paths = []
    num_rec_csv = sum(1 for line in csvfile)
    coords = np.zeros((num_rec_csv-1, 4))  # Declare a array of zeros
    reader = csv.reader(csvfile, delimiter=',')  # Read file using appropriated delimiter
    csvfile.seek(0)
    for i, row in enumerate(reader, start=0):
        if i == 0:  # First row is
            continue
        # Read row values
        path, width, height, _, x0, y0, x1, y1 = row
        path = "./images/" + path
        
        # Since first row was skipped, therefore reducing index by 1
        coords[i-1, 0] = int(x0) * IMAGE_SIZE / int(width) # Normalize bounding box by image size
        coords[i-1, 1] = int(y0) * IMAGE_SIZE / int(height) # Normalize bounding box by image size
        coords[i-1, 2] = (int(x1) - int(x0)) * IMAGE_SIZE / int(width) # Normalize bounding box by image size
        coords[i-1, 3] = (int(y1) - int(y0)) * IMAGE_SIZE / int(height)
        paths.append(path)

Import MobileNet and load MobileNet into a variable named 'model' which takes input shape of 128 128 3. Freeze all the layers. Add convolution and reshape layers at the end to ensure the output is 4 coordinates¶

ALPHA = 1.0 # Width hyper parameter for MobileNet (0.25, 0.5, 0.75, 1.0). Higher width means more accurate but slower

model = MobileNet(input_shape=(IMAGE_SIZE, IMAGE_SIZE, 3), include_top=False, alpha=ALPHA) # Load pre-trained mobilenet
# Do not include classification (top) layer

# to freeze layers, except the new top layer, of course, which will be added below
for layer in model.layers:
    layer.trainable = False

# Add new top layer which is a conv layer of the same size as the previous layer so that only 4 coords of BBox can be output
x = model.layers[-1].output
x = Conv2D(4, kernel_size=4, name="coords")(x)
# In the line above kernel size should be 3 for img size 96, 4 for img size 128, 5 for img size 160 etc.
x = Reshape((4,))(x) # These are the 4 predicted coordinates of one BBox

model = Model(inputs=model.input, outputs=x)

Define a custom loss function IoU which calculates Intersection Over Union¶

In [9]:
def loss(gt,pred):
    intersections = 0
    unions = 0
    diff_width = np.minimum(gt[:,0] + gt[:,2], pred[:,0] + pred[:,2]) - np.maximum(gt[:,0], pred[:,0])
    diff_height = np.minimum(gt[:,1] + gt[:,3], pred[:,1] + pred[:,3]) - np.maximum(gt[:,1], pred[:,1])
    intersection = diff_width * diff_height
    
    # Compute union
    area_gt = gt[:,2] * gt[:,3]
    area_pred = pred[:,2] * pred[:,3]
    union = area_gt + area_pred - intersection

#     Compute intersection and union over multiple boxes
    for j, _ in enumerate(union):
        if union[j] > 0 and intersection[j] > 0 and union[j] >= intersection[j]:
            intersections += intersection[j]
            unions += union[j]

    # Compute IOU. Use epsilon to prevent division by zero
    iou = np.round(intersections / (unions + epsilon()), 4)
    iou = iou.astype(np.float32)
    return iou

def IoU(y_true, y_pred):
    #iou = tf.py_func(loss, [y_true, y_pred], tf.float32)
    iou = tf.py_function(loss, [y_true, y_pred], tf.float32)
    return iou

#window - how long the sequence will be
def create_dataset(dataset, dataset_y, window=1):
    
    dataX, dataY = [], []
    
    for i in range(len(dataset)-window):
        
        #a = dataset[i:(i+window), 0]  # Pick only first(index=0) column
        a = dataset[i:(i+window), ]  # Pick every column
        dataX.append(a)
        #dataY.append(dataset[i + window, 0])  # store close of the next row as the target for this row
        """
        While calculation y below, if 0 is not used then the output has shape like (1979, ).
        If 0 is used, then the output has shape like (1979, 1).
        """
        dataY.append(dataset_y[i + window, 0])  # store close of the next row as the target for this row
    
    return np.array(dataX), np.array(dataY)
